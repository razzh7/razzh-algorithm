## 思想
每次循环都取最小的数放到左边的有序序列中
```js
let arr = [8,1,6,3,21,2]
function selectSort(arr) {
  for(let i = 0; i < arr.length - 1; i++) { // -1: 防止数组越界
    let minIndex = i; // 取最小值
    for(let j = i + 1; j < arr.length; j++) { // j = i + 1:每次排序都从左边的最后一个开始
      if(arr[j] < arr[minIndex]) {
        minIndex = j;
      }
    }
    let temp = arr[minIndex];
        arr[minIndex] = arr[i];
        arr[i] = temp;
  }
  return arr;
}
```
